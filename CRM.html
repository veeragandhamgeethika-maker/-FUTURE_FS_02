const express = require('express'); const mongoose = require('mongoose'); const cors = require('cors'); const app = express(); const PORT = 5000; const ADMIN_KEY = 'admin123'; // Change this for security app.use(cors()); app.use(express.json()); app.use(express.urlencoded({ extended: true })); // 1. MongoDB Setup mongoose.connect('mongodb://127.0.0.1:27017/minicrm', { useNewUrlParser: true, useUnifiedTopology: true }).then(() => console.log("MongoDB Connected")) .catch(err => console.log("MongoDB Error:", err)); const followupSchema = new mongoose.Schema({ note: String, date: {type: Date, default: Date.now} }); const leadSchema = new mongoose.Schema({ name: {type: String, required: true}, email: {type: String, required: true, unique: true}, source: {type: String, default: 'Website'}, status: {type: String, enum: ['new', 'contacted', 'converted'], default: 'new'}, notes: {type: String, default: ''}, followups: [followupSchema], createdAt: {type: Date, default: Date.now} }); const Lead = mongoose.model('Lead', leadSchema); // 2. Admin Middleware - Secure access const checkAdmin = (req, res, next) => { if(req.headers['x-admin-key']!== ADMIN_KEY && req.query.key!== ADMIN_KEY) return res.status(401).send('
